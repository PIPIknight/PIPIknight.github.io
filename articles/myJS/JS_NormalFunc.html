防抖 debounce #

防抖函数几个注意点，timer，fn的this指向？，以及防抖函数对额外参数的处理

js
// 额外参数也可设置在这里 debounce = (fn, delay, ...args)
const debounce = (fn, delay) => {
    // 迭代器
    let timer;
    // 额外参数有多种处理方式，可随机应变
    return function (...args) {
        if (timer) clearTimeout(timer);
        timer = setTimeout(() => {
            console.log("args==", args);
            fn.apply(this, args);
        }, delay);
    };
};

const consoleTask = (text) => {
    console.log("text==", text);
};

const task = debounce(consoleTask, 1000);

// 注意对额外参数的处理
window.addEventListener("mousemove", task.bind(this, '888'));

节流 throttle #

时间戳方案 （ fn会先执行，之后的delay时间内都不能再触发fn ）

js
function throttle(fn, delay) {
    // 时间戳
    let last = 0;
    return function (...args) {
        const now = Date.now();
        // 节流时间段前执行
        if (now - last > delay) {
            last = now;
            fn.apply(this, args);
        }
    };
}

const task = () => console.log("run");
const throttleTask = throttle(task, 1000);

window.addEventListener("mousedown", throttleTask);

定时器方案 （ fn会在delay时间之后执行，delay期间无法再触发 ）

js
function throttle(fn, delay) {
    // 定时器
    let timer = null;
    return function (...args) {
        if (!timer) {
            timer = setTimeout(() => {
                // 节流时间段后执行
                fn.apply(this, args);
                timer = null;
            }, delay);
        }
    };
}

const task = () => console.log("run");
const throttleTask = throttle(task, 1000);

window.addEventListener("mousedown", throttleTask);

简单的对象递归拷贝 #

js
function deepCloneObj(obj, cache = new WeakMap()) {
    if (obj === null || typeof obj !== "object") return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof RegExp) return new RegExp(obj);

    if (cache.has(obj)) return cache.get(obj); // 如果出现循环引用，则返回缓存的对象，防止递归进入死循环
    let cloneObj = new obj.constructor(); // 使用对象所属的构造函数创建一个新对象那个
    cache.set(obj, cloneObj); // 缓存对象，用于循环引用的情况

    for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
            cloneObj[key] = deepCloneObj(obj[key], cache); // 递归拷贝
        }
    }

    return cloneObj;
}

const obj = { name: "Jack", address: { x: 100, y: 20 } };
// obj.a = obj  // 循环引用
const newObj = deepCloneObj(obj);
console.log(newObj);
console.log(newObj === obj);

继承ES5与ES6版 #

ES5：寄生组合继承，实质是先创建子类的实例对象，再将父类的方法添加到this上

js
function Parent(name) {
    this.name = name;
}

Parent.prototype.sayHi = function () {
    console.log(this.name + " Hi!");
};

function Son(name, age) {
    this.age = age;
    Parent.call(this, name); // 将父类的方法添加到this上
}
// Object.create() 方法用于创建一个新对象，使用现有的对象来作为新创建对象的原型（prototype）。
Son.prototype = Object.create(Parent.prototype); // 以Parent的原型为对象创建一个新对象，继承Parent的方法
Son.prototype.contructor = Son; // 构造函数重新指向Son

let wb = new Son("wangbin", 27);
console.log(wb.name);
console.log(wb.age);
wb.sayHi();

ES6：ES6的继承实质是先创建父类的实例对象this，再由子类的构造函数修改父类的this

js
class Parent {
  constructor(name) {
    this.name = name
  }
  sayHi() {
    console.log(this.name + ' Hi!')
  }
}

class Son extends Parent {
  constructor(name, age) {
    // 创建父类的实例对象this,这里调用了super方法！super作为方法和对象的意义也是不同的
    super(name)
    this.age = age
  }
}

数组转树！ #

常规方法，递归（ 缺点，数据量大时会很慢 ）

js
let arr = [
    { id: 1, name: "部门1", pid: 0 },
    { id: 2, name: "部门2", pid: 1 },
    { id: 3, name: "部门3", pid: 1 },
    { id: 4, name: "部门4", pid: 3 },
    { id: 5, name: "部门5", pid: 4 },
];

function getChildren(data, result, pid) {
    for (let item of data) {
        if (item.pid == pid) {
            const newItem = { ...item, children: [] };
            result.push(newItem);

            getChildren(data, newItem.children, newItem.id);
        }
    }
}
const arrayToTree = (data, pid) => {
    const result = [];
    getChildren(data, result, pid);
    return result;
};
console.log(arrayToTree(arr, 0));

重点方法：hashmap，核心思想是空间换时间

js
let arr = [
    { id: 1, name: "部门1", pid: 0 },
    { id: 2, name: "部门2", pid: 1 },
    { id: 3, name: "部门3", pid: 1 },
    { id: 4, name: "部门4", pid: 3 },
    { id: 5, name: "部门5", pid: 4 },
];
function arrayToTree(items) {
    const result = [];
    const itemMap = {};
    for (const item of items) {
        const id = item.id;
        const pid = item.pid;
        // 最后 itemMap[1]处一定会生成完整的一棵树，其他位置会生成以自身为根节点的子树
        if (!itemMap[id]) {
            itemMap[id] = {
                children: [],
            };
        }
        itemMap[id] = {
            ...item,
            children: itemMap[id]["children"],
        };
        const treeItem = itemMap[id];
        if (pid === 0) {
            result.push(itemMap[id]);
        } else {
            if (!itemMap[pid]) {
                itemMap[pid] = {
                    children: [],
                };
            }
            itemMap[pid].children.push(treeItem);
        }
    }
    return itemMap;
}
console.log(arrayToTree(arr));
